298 {
299 tamanoSeleccionado = comboTamano->SelectedIndex;
300 twoPlayerMode = (comboJugadores->SelectedIndex == 1);
301
302 int flags = 0;
303 if (chkObstacles->Checked) flags |= static_cast<int>(GameMode::Obstacles);
304 if (chkRandomGrowth->Checked) flags |= static_cast<int>(GameMode::RandomGrowth);
305 if (chkSpeedIncrease->Checked) flags |= static_cast<int>(GameMode::SpeedIncrease);
306 GameMode combinedMode = static_cast<GameMode>(flags);
307
308 if (gameTimer != nullptr) { gameTimer->Stop(); gameTimer = nullptr; }
309
310 game1 = CreateGame(combinedMode);
311 game1->OnGameOver += gcnew SnakeGame::GameOverDelegate(this, &Form1::Game1_OnGameOver);
312 game1->OnScoreChanged += gcnew SnakeGame::ScoreChangedDelegate(this, &Form1::Game1_OnScoreChanged);
313 game1->OnSpeedChanged += gcnew SnakeGame::SpeedChangedDelegate(this, &Form1::Game_OnSpeedChanged);

315 if (twoPlayerMode) {

317 game2->OnGameOver += gcnew SnakeGame::GameOverDelegate(this, &Form1::Game2_OnGameOver);
318 game2->OnScoreChanged += gcnew SnakeGame::ScoreChangedDelegate(this, &Form1::Game2_OnScoreChanged);
319 game2->OnSpeedChanged += gcnew SnakeGame::SpeedChangedDelegate(this, &Form1::Game_OnSpeedChanged);
320 }
321 else {
322 game2 = nullptr;
323 }
324
325 lblScore1->Text = L"J1: 0 pts";
326 lblScore2->Text = L"J2: 0 pts";
327 lblHighScore->Text = String::Format(L"Record: {0}", game1->HighScore);

329 lblP1Tag->Visible = true;

418 void CheckBothDead()
419 {
420 bool g1Dead = (game1 == nullptr || game1->IsGameOver);
421 bool g2Dead = !twoPlayerMode || (game2 == nullptr || game2->IsGameOver);
422
423 if (!g1Dead || !g2Dead) return;
424
425 gameTimer->Stop();
426
427 if (twoPlayerMode) {
428 int s1 = (game1 != nullptr) ? game1->Score : 0;
429 int s2 = (game2 != nullptr) ? game2->Score : 0;
430
431 String^ winnerMsg;
432 if (s1 > s2) winnerMsg = String::Format(L"🏆 ¡Ganó el Jugador 1! ({0} pts)", s1);
433 else if (s2 > s1) winnerMsg = String::Format(L"🏆 ¡Ganó el Jugador 2! ({0} pts)", s2);
434 else winnerMsg = L"🤝 ¡Empate!";
435
436 MessageBox::Show(winnerMsg, L"Resultado final",
437 MessageBoxButtons::OK, MessageBoxIcon::Information);
438 }
439
440 if (game1 != nullptr) {
441 String^ user1 = PromptUsernameFor(L"Jugador 1");
442 SaveScore(user1, game1->Score, game1);
443 }
444
445 if (twoPlayerMode && game2 != nullptr) {
446 String^ user2 = PromptUsernameFor(L"Jugador 2");
447 SaveScore(user2, game2->Score, game2);
448 }
449
450 lblHighScore->Text = String::Format(L"Record: {0}", ScoreManager::GetHighestScore());
451
452 panelGame1->Invalidate();
453 if (twoPlayerMode) panelGame2->Invalidate();
454 }

504 void Form1_KeyDown(Object^ sender, KeyEventArgs^ e)
505 {
506 if (game1 != nullptr && !game1->IsGameOver) {
507 switch (e->KeyCode) {
508 case Keys::Up: game1->SetDirection(0, -1); break;
509 case Keys::Down: game1->SetDirection(0, 1); break;
510 case Keys::Left: game1->SetDirection(-1, 0); break;
511 case Keys::Right: game1->SetDirection(1, 0); break;
512 default: break;
513 }
514 }
515
516 if (twoPlayerMode && game2 != nullptr && !game2->IsGameOver) {
517 switch (e->KeyCode) {
518 case Keys::W: game2->SetDirection(0, -1); break;
519 case Keys::S: game2->SetDirection(0, 1); break;
520 case Keys::A: game2->SetDirection(-1, 0); break;
521 case Keys::D: game2->SetDirection(1, 0); break;
522 default: break;
523 }
524 }
525
526 if (e->KeyCode == Keys::P) {
527 bool isPaused = (game1 != nullptr && game1->IsPaused);
528 if (game1 != nullptr && !game1->IsGameOver) game1->TogglePause();
529 if (game2 != nullptr && !game2->IsGameOver) game2->TogglePause();
530 panelGame1->Invalidate();
531 if (twoPlayerMode) panelGame2->Invalidate();
532 }
533
534 if (e->KeyCode == Keys::R) {
535 bool anyOver = (game1 != nullptr && (game1->IsGameOver || game1->HasStarted));
536 if (anyOver) {
537 if (game1 != nullptr) { game1->Reset(); game1->SetSnakeSkin(selectedSnakePrimaryColor, selectedSnakeSecondaryColor, selectedSnakeHeadColor); }
538 if (game2 != nullptr) { game2->Reset(); game2->SetSnakeSkin(selectedSnakePrimaryColor, selectedSnakeSecondaryColor, selectedSnakeHeadColor); }
539 if (gameTimer != nullptr) {
540 gameTimer->Interval = game1->SpeedMs;
541 if (!gameTimer->Enabled) gameTimer->Start();
542 }
543 lblScore1->Text = L"J1: 0 pts";
544 lblScore2->Text = L"J2: 0 pts";
545 panelGame1->Invalidate();
546 if (twoPlayerMode) panelGame2->Invalidate();
547 }
548 }
549
550 if (e->KeyCode == Keys::M) {
551 bool g1Done = (game1 == nullptr || game1->IsGameOver);
552 bool g2Done = !twoPlayerMode || (game2 == nullptr || game2->IsGameOver);
553 if (g1Done && g2Done) {
554 if (gameTimer != nullptr) gameTimer->Stop();
555 panelGame1->Visible = false;
556 panelGame2->Visible = false;
557 lblP1Tag->Visible = false;
558 lblP2Tag->Visible = false;
559 lblScore1->Visible = false;
560 lblScore2->Visible = false;
561 lblHighScore->Visible = false;
562 panelMenu->Visible = true;
563 }
564 }
565
566 if (e->KeyCode == Keys::L) {
567 bool g1Done = (game1 == nullptr || game1->IsGameOver);
568 bool g2Done = !twoPlayerMode || (game2 == nullptr || game2->IsGameOver);
569 if (g1Done && g2Done) {
570 LeaderboardForm^ lb = gcnew LeaderboardForm();
571 lb->ShowDialog(this);
572 }
573 }
