145 void LoadLeaderboard()
146 {
147 gridScores->Rows->Clear();
148
149 auto topScores = ScoreManager::GetTopScores(10);
150 for (int i = 0; i < topScores->Count; i++)
151 {
152 ScoreEntry^ entry = topScores[i];
153 gridScores->Rows->Add(
154 (i + 1).ToString(),
155 entry->Username,
156 entry->Score.ToString(),
157 entry->BestScore.ToString(),
158 entry->BoardSize,
159 entry->GameMode);
160 }
161
162 if (topScores->Count == 0)
163 gridScores->Rows->Add(L"-", L"Sin registros", L"0", L"0", L"-", L"-");
164 }
