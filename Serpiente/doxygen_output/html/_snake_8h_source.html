72 void Generate(Random^ rnd, int cols, int rows, List<GridPoint>^ occupiedCells) {
73 int minX = 1, maxX = cols - 1;
74 int minY = 1, maxY = rows - 1;
75 if (maxX <= minX) maxX = minX + 1;
76 if (maxY <= minY) maxY = minY + 1;
77
78 for (int attempts = 0; attempts < 100; attempts++) {
79 GridPoint newPos(rnd->Next(minX, maxX), rnd->Next(minY, maxY));
80 bool isOccupied = false;
81 for each (GridPoint p in occupiedCells) {
82 if (p.Equals(newPos)) { isOccupied = true; break; }
83 }
84 if (!isOccupied) { Position = newPos; return; }
85 }
86 Position = GridPoint(rnd->Next(minX, maxX), rnd->Next(minY, maxY));
87 }

235 void Tick() {
236 if (IsGameOver || IsPaused) return;
237 HasStarted = true;
238 dx = pendingDx; dy = pendingDy;
239
240 GridPoint head = snakeBody[0];
241 GridPoint newHead(head.X + dx, head.Y + dy);
242
243 if (WallPassThrough) {
244 if (newHead.X <= 0) newHead.X = CurrentCols - 2;
245 else if (newHead.X >= CurrentCols - 1) newHead.X = 1;
246 if (newHead.Y <= 0) newHead.Y = CurrentRows - 2;
247 else if (newHead.Y >= CurrentRows - 1) newHead.Y = 1;
248 }
249 else {
250 for each (Wall ^ w in walls) {
251 if (w->Collides(newHead)) { TriggerGameOver(); return; }
252 }
253 }
254
255 for (int i = 0; i < snakeBody->Count - 1; i++) {
256 if (snakeBody[i].Equals(newHead)) { TriggerGameOver(); return; }
257 }
258
259 snakeBody->Insert(0, newHead);
260 if (growPending > 0) growPending--;
261 else snakeBody->RemoveAt(snakeBody->Count - 1);
262
263 if (newHead.Equals(apple->Position)) {
264 ApplesEaten++;
265 Score += SCORE_PER_APPLE;
266
267 int grow = GrowAmount;
268 // RandomGrowth activo → crecimiento aleatorio
269 if (HasMode(GameMode::RandomGrowth))
270 grow = rnd->Next(1, 6);
271
272 growPending += grow;
273 apple->Generate(rnd, CurrentCols, CurrentRows, GetOccupiedCells());
274 OnScoreChanged(Score, ApplesEaten);
275 UpdateByScore();
276 }

323 List<int>^ LoadScores() {
324 List<int>^ scores = gcnew List<int>();
325 if (!System::IO::File::Exists(ScoreConfig::FILE_NAME)) return scores;
326 try {
327 System::IO::StreamReader^ sr = gcnew System::IO::StreamReader(ScoreConfig::FILE_NAME);
328 String^ line;
329 while ((line = sr->ReadLine()) != nullptr) {
330 int val;
331 array<String^>^ parts = line->Split(',');
332 if (int::TryParse(line->Trim(), val)) scores->Add(val);
333 else if (parts->Length >= 2 && int::TryParse(parts[1]->Trim(), val)) scores->Add(val);
334 }
335 sr->Close();
336 }
337 catch (...) {}
338 return scores;
339 }

367 void UpdateByScore() {
368 // Obstacles activo → generar obstáculo cada umbral
369 if (HasMode(GameMode::Obstacles)) {
370 int currentBlock = Score / OBSTACLE_SCORE_THRESHOLD;
371 int lastBlock = (lastObstacleScore >= 0)
372 ? lastObstacleScore / OBSTACLE_SCORE_THRESHOLD : -1;
373 if (currentBlock > lastBlock) {
374 GenerateObstacle();
375 lastObstacleScore = Score;
376 }
377 }
378
379 // SpeedIncrease activo → reducir intervalo cada umbral
380 if (HasMode(GameMode::SpeedIncrease)) {
381 int currentBlock = Score / SPEED_SCORE_THRESHOLD;
382 int lastBlock = (lastSpeedScore >= 0)
383 ? lastSpeedScore / SPEED_SCORE_THRESHOLD : -1;
384 if (currentBlock > lastBlock) {
385 int prevSpeed = SpeedMs;
386 SpeedMs = Math::Max((int)(SpeedMs * 0.90), MIN_SPEED_MS);
387 lastSpeedScore = Score;
388 if (SpeedMs != prevSpeed) OnSpeedChanged(SpeedMs);
389 }
390 }
391 }
